event CompetitiveGame_LowerChatWindow
{
	Animate HudChat ypos 360 Accel 0 0
}
//r140

event CompetitiveGame_RestoreChatWindow // GAME CHAT
{
	Animate HudChat xpos -7 Accel 0 0
	Animate HudChat ypos r160 Accel 0 0
}
//275

event HudTournament_MoveChatWindow
{
	RunEvent CompetitiveGame_LowerChatWindow 0
	RunEvent CompetitiveGame_RestoreChatWindow 0
}

event HudTournament_ShowTimerDefault
{
	Animate CountdownLabel ypos 366 Linear 0 0
	Animate CountdownLabelShadow ypos 366 Linear 0 0

	SetVisible CountdownBG  1 0
	SetVisible CountdownLabel 1 0
	SetVisible CountdownLabelShadow 1 0
}

event HudReadyPulse
{
	Animate	TournamentInstructionsLabel 	FgColor		"colGenRed"		Linear 0.0 0.0
	Animate	TournamentInstructionsLabel 	FgColor		"colGenRed"		Linear 0.0 0.0

	RunEvent HudReadyPulseLoop	10.0
}

event HudReadyPulseLoop
{
	RunEvent HudReadyPulse			0.0
}

event HudReadyPulseEnd
{
	Animate	TournamentInstructionsLabel 	FgColor		"colGenGreen"		Linear 0.0 0.0
	
	StopEvent HudReadyPulse			0.0
	StopEvent HudReadyPulseLoop		0.0
}
